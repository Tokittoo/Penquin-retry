        // Header scroll behavior
        let lastScroll = 0;
        const header = document.querySelector('header');

        window.addEventListener('scroll', () => {
            const currentScroll = window.pageYOffset;
            if (currentScroll <= 0) {
                header.classList.remove('hidden');
                return;
            }

            if (currentScroll > lastScroll && !header.classList.contains('hidden')) {
                header.classList.add('hidden');
            } else if (currentScroll < lastScroll && header.classList.contains('hidden')) {
                header.classList.remove('hidden');
            }
            lastScroll = currentScroll;
        });

        // Theme handling
        const themeToggle = document.querySelector('.theme-toggle');
        const body = document.body;

            body.setAttribute('data-theme', isDark ? 'dark' : 'light');
            icon.className = isDark ? 'fas fa-sun' : 'fas fa-moon';
            localStorage.setItem('theme', isDark ? 'dark' : 'light');
        }

        themeToggle.addEventListener('click', () => {
            const isDark = body.getAttribute('data-theme') !== 'light';
            setTheme(!isDark);
        });

        // Initialize theme from localStorage
        const savedTheme = localStorage.getItem('theme') || 'dark';

        document.getElementById('domainForm').addEventListener('submit', function(e) {

            if (domain) {

                showNotification('Commands updated for: ' + domain);
            }
        });

        // Update all commands with the new domain
        function updateAllCommands(domain) {
            const toolsContainer = document.querySelector('.tools-container');
            toolsContainer.classList.add('fade');
            
            setTimeout(() => {
                for (const [id, template] of Object.entries(commandTemplates)) {
                    const updatedCommand = template.replace(/example\.com/g, domain)
                                                 .replace(/site\.com/g, domain)
                                                 .replace(/target\.com/g, domain)
                                                 .replace(/domain/g, domain);
                    document.getElementById(id).textContent = updatedCommand;
                }
                toolsContainer.classList.remove('fade');
                
                // Make sure to render extensions after domain update
                renderExtensions();
            }, 300);
        }

                let fixedCommandText = commandText
                    // Handle line breaks properly by replacing <br> with actual newlines
                    .replace(/<br>/g, '\n')
                    // Ensure no double spaces
                    .replace(/\s{2,}/g, ' ')
                    // Fix spacing issue with cat/echo/grep commands - ensure proper spacing
                    .replace(/([^\s\n])(cat|echo|grep|sed|awk|cut|sort|uniq|tee|xargs|curl)\s/g, '$1 $2 ')
                    // Fix issue with pipe symbols - ensure proper spacing around pipes
                    .replace(/([^\s\n])\|([^\s\n])/g, '$1 | $2')
                    .replace(/([^\s\n])\|(\s)/g, '$1 |$2')
                    .replace(/(\s)\|([^\s\n])/g, '$1| $2')
                    // Add space before redirection operators if missing
                    .replace(/([^\s\n])(\>|\>\>)/g, '$1 $2')
                    // Fix spacing for common tools
                    .replace(/([^\s\n])(katana|httpx|gau|nuclei|ffuf|dalfox|waybackurls|gf|Gxss|urldedupe|anew)\s/g, '$1 $2 ');
                
                // Fix edge case of a newline followed immediately by a command
                fixedCommandText = fixedCommandText.replace(/\n(cat|echo|grep|sed|awk|cut|sort|uniq|tee|xargs|curl|httpx|gau|nuclei|ffuf|dalfox)/g, '\n$1');
                
                // Ensure each line starts with proper spacing if it begins with a command
                fixedCommandText = fixedCommandText.split('\n').map(line => {
                    // Trim leading/trailing whitespace from each line
                    line = line.trim();
                    return line;
                }).join('\n');
                
                await navigator.clipboard.writeText(fixedCommandText);
                button.classList.add('copied');
                showNotification();
                setTimeout(() => button.classList.remove('copied'), 500);

                showNotification('Failed to copy command', true);
            }

        // Create trail elements
        function createTrails() {
            for (let i = 0; i < cursorTrailCount; i++) {
                const trail = document.createElement('div');
                trail.className = 'cursor-trail';
                // Update to green color scheme
                trail.style.backgroundColor = `hsl(${160 + i*2}, 100%, 50%)`; 
                document.body.appendChild(trail);
                trails.push({
                    element: trail,
                    x: 0,
                    y: 0,
                    size: 10 - (i * 0.3), // Gradually decrease size
                    alpha: 1 - (i * 0.03), // Gradually decrease opacity
                    speed: 0.15 - (i * 0.005) // Different speeds for each particle
                });

        let lastY = 0;
        document.addEventListener('mousemove', (e) => {
            mouseX = e.clientX;
            mouseY = e.clientY;
            
            // Update cursor dot position (follows exactly)
            cursorDot.style.left = `${mouseX}px`;
            cursorDot.style.top = `${mouseY}px`;
        });

        // Animate the trails with improved physics
        function animateTrails() {
            // Smooth trail movement
            for (let i = 0; i < trails.length; i++) {
                const trail = trails[i];
                
                // Calculate trail position with easing
                const targetX = i === 0 ? mouseX : trails[i-1].x;
                const targetY = i === 0 ? mouseY : trails[i-1].y;
                
                // Smoothed movement
                trail.x += (targetX - trail.x) * trail.speed;
                trail.y += (targetY - trail.y) * trail.speed;
                
                // Apply position
                trail.element.style.left = `${trail.x}px`;
                trail.element.style.top = `${trail.y}px`;
                
                // Set size proportional to position in trail
                const size = Math.max(4, trail.size);
                trail.element.style.width = `${size}px`;
                trail.element.style.height = `${size}px`;
                
                // Set opacity
                const distance = Math.sqrt(
                    Math.pow(mouseX - trail.x, 2) + 
                    Math.pow(mouseY - trail.y, 2)
                );
                
                const opacity = Math.max(0, trail.alpha - (distance * 0.003));
                trail.element.style.opacity = opacity.toString();
            }
            
            requestAnimationFrame(animateTrails);
        }

                const mouseY = e.clientY / window.innerHeight - 0.5;
                
                cards.forEach(card => {
                    const rect = card.getBoundingClientRect();
                    const cardX = rect.left + rect.width / 2;
                    const cardY = rect.top + rect.height / 2;
                    
                    // Calculate distance from mouse to card center (0 to 1)
                    const distX = Math.abs(e.clientX - cardX) / window.innerWidth;
                    const distY = Math.abs(e.clientY - cardY) / window.innerHeight;
                    const dist = Math.sqrt(distX * distX + distY * distY);
                    
                    // Only apply effect to cards near the mouse
                    if (dist < 0.3) {
                        const intensity = (1 - dist / 0.3) * 5; // Stronger effect for closer cards
                        card.style.transform = `perspective(1000px) rotateY(${mouseX * intensity}deg) rotateX(${-mouseY * intensity}deg) translateZ(10px)`;

        // Function to create and control the matrix background effects
        function initMatrixBackground() {
            const particlesContainer = document.querySelector('.particles-container');
            const lightFlare = document.querySelector('.light-flare');
            const particleCount = 50; // Number of floating particles
            
            // Create particles
            for (let i = 0; i < particleCount; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                
                // Random size between 2-6px
                const size = Math.random() * 4 + 2;
                particle.style.width = `${size}px`;
                particle.style.height = `${size}px`;
                
                // Random position
                particle.style.left = `${Math.random() * 100}%`;
                particle.style.top = `${Math.random() * 100}%`;
                
                // Random opacity
                particle.style.opacity = (Math.random() * 0.5 + 0.1).toString();
                
                // Random animation properties
                const duration = Math.random() * 40 + 20; // 20-60s
                const delay = Math.random() * 10;
                
                // Apply floating animation
                particle.style.animation = `particleFloat ${duration}s ease-in-out ${delay}s infinite alternate`;
                
                particlesContainer.appendChild(particle);
            }
            
            // Light flare following mouse
            document.addEventListener('mousemove', (e) => {
                lightFlare.style.opacity = '1';
                lightFlare.style.left = `${e.clientX - 100}px`;
                lightFlare.style.top = `${e.clientY - 100}px`;
                
                // Gradually fade out if mouse stops moving
                clearTimeout(window.flareTimeout);
                window.flareTimeout = setTimeout(() => {
                    lightFlare.style.opacity = '0';
                }, 2000);
            });
            
            // Occasionally add pulsing effect to random particles
            setInterval(() => {
                const particles = document.querySelectorAll('.particle');
                const randomIndex = Math.floor(Math.random() * particles.length);
                const particle = particles[randomIndex];
                
                // Add pulse effect
                particle.style.transform = 'scale(3)';
                particle.style.opacity = '0.8';
                particle.style.filter = 'blur(2px)';
                
                // Reset after animation
                setTimeout(() => {
                    particle.style.transform = 'scale(1)';
                    particle.style.opacity = (Math.random() * 0.5 + 0.1).toString();
                    particle.style.filter = 'none';
                }, 1000);
            }, 2000);
        }

        // Mobile menu toggle
        document.addEventListener('DOMContentLoaded', function() {
            const menuToggle = document.querySelector('.menu-toggle');
            const navLinks = document.querySelector('.nav-links');
            const mobileNavOverlay = document.querySelector('.mobile-nav-overlay');
            
            // Function to open mobile menu
            function openMobileMenu() {
                navLinks.classList.add('active');
                menuToggle.classList.add('active');
                if (mobileNavOverlay) {
                    mobileNavOverlay.classList.add('active');
                }
                document.body.style.overflow = 'hidden';
            }
            
            // Function to close mobile menu
            function closeMobileMenu() {
                navLinks.classList.remove('active');
                menuToggle.classList.remove('active');
                if (mobileNavOverlay) {
                    mobileNavOverlay.classList.remove('active');
                }
                document.body.style.overflow = '';
            }
            
            if (menuToggle && navLinks) {
                // Toggle menu when menu button is clicked
                menuToggle.addEventListener('click', function(e) {
                    e.stopPropagation(); // Prevent click from bubbling to document
                    if (navLinks.classList.contains('active')) {
                        closeMobileMenu();
                    } else {
                        openMobileMenu();
                    }
                });
                
                // Close menu when a link is clicked
                const navItems = navLinks.querySelectorAll('a');
                navItems.forEach(item => {
                    item.addEventListener('click', function() {
                        closeMobileMenu();
                    });
                });
                
                // Handle the X close button click (added via ::after pseudo-element)
                navLinks.addEventListener('click', function(e) {
                    // Check if click was on the close button area (top right of the menu)
                    const rect = navLinks.getBoundingClientRect();
                    const closeArea = { 
                        x: rect.right - 50, 
                        y: rect.top + 20, 
                        width: 30, 
                        height: 30 
                    };
                    
                    if (
                        e.clientX >= closeArea.x && 
                        e.clientX <= closeArea.x + closeArea.width &&
                        e.clientY >= closeArea.y && 
                        e.clientY <= closeArea.y + closeArea.height
                    ) {
                        closeMobileMenu();
                    }
                });
            }
            
            // Close menu when clicking the overlay
            if (mobileNavOverlay) {
                mobileNavOverlay.addEventListener('click', function() {
                    closeMobileMenu();
                });
            }
            
            // Close menu when clicking outside
            document.addEventListener('click', function(event) {
                if (!event.target.closest('.navbar') && !event.target.closest('.nav-links')) {
                    if (navLinks.classList.contains('active')) {
                        closeMobileMenu();
                    }
                }
            });
            
            // Close menu with Escape key
            document.addEventListener('keydown', function(e) {
                if (e.key === 'Escape' && navLinks.classList.contains('active')) {
                    closeMobileMenu();
                }
            });
            
            // Fix for iOS devices - ensure menu closes when links are clicked
            if (/iPhone|iPad|iPod/i.test(navigator.userAgent)) {
                document.querySelectorAll('.nav-links a').forEach(link => {
                    link.addEventListener('touchend', function(e) {
                        setTimeout(closeMobileMenu, 300);
                    });
                });
            }
        });

        // Methodology loading and display logic
        async function loadMethodologies() {
            try {
                const indexResponse = await fetch('Lostsec-methodlogy-main/methodologies.json');
                if (!indexResponse.ok) {
                    throw new Error(`HTTP error! status: ${indexResponse.status}`);
                }
                
                const methodologyIndex = await indexResponse.json();
                
                if (!methodologyIndex.methodologies || !Array.isArray(methodologyIndex.methodologies)) {
                    throw new Error('Invalid methodology data structure');
                }

                renderMethodologyCards(methodologyIndex.methodologies);

                const methodologyContent = document.getElementById('methodology-content');
                methodologyContent.addEventListener('click', async (e) => {
                    const card = e.target.closest('.methodology-card');
                    if (card) {
                        try {
                            const methodologyId = card.dataset.id;
                            const methodology = methodologyIndex.methodologies.find(m => m.id === methodologyId);
                            if (!methodology) {
                                throw new Error(`Methodology not found: ${methodologyId}`);
                            }

                            const response = await fetch(`Lostsec-methodlogy-main/${methodology.path}`);
                            if (!response.ok) {
                                throw new Error(`Failed to load methodology: ${methodology.path}`);
                            }

                            const data = await response.json();
                            showMethodology(data);
                        } catch (err) {
                            console.error('Error loading methodology:', err);
                        }
                    }

                    if (e.target.closest('.back-btn')) {
                        renderMethodologyCards(methodologyIndex.methodologies);
                    }
                });
            } catch (error) {
                console.error('Methodology initialization error:', error);
                const container = document.getElementById('methodology-content');
                container.innerHTML = `
                    <div class="error-message">
                        <i class="fas fa-exclamation-circle"></i>
                        <p>Sorry, we couldn't load the methodologies. Please check the file paths.</p>
                    </div>
                `;
            }
        }

            const container = document.getElementById('methodology-content');

                            </div>

                                <h3>${methodology.title}</h3>
                            </div>

                                    </span>
                                `).join('')}
                            </div>
                            <div class="card-footer">
                                <i class="fas fa-terminal"></i><span>View Methodology</span>
                                <i class="fas fa-arrow-right"></i>
                            </div>
                        </div>
                    `).join('')}
                </div>
            `;
        }

        function showMethodology(methodology) {
            const container = document.getElementById('methodology-content');
            const escapeHtml = (str) => {
                const div = document.createElement('div');
                div.textContent = str;
                return div.innerHTML;
            };

            // Default rendering for methodologies
            const hasCommands = methodology.commands && Array.isArray(methodology.commands);

            container.innerHTML = `
                <div class="methodology-detail">
                    <button class="back-btn">
                        <i class="fas fa-arrow-left"></i> Back
                    </button>
                    <h2 class="methodology-title">
                        <i class="fas fa-code"></i> ${escapeHtml(methodology.title || 'Untitled')}
                    </h2>
                    <div class="methodology-category">${escapeHtml(methodology.category || 'Security')}</div>
                    
                    <div class="commands-section">
                        ${hasCommands ? methodology.commands.map(cmd => `
                            <div class="methodology-command-block">
                                <div class="terminal-box">
                                    <span class="prompt">$</span>
                                    <span class="command">${escapeHtml(cmd.cmd || '')}</span>
                                    <button class="copy-btn methodology-copy-btn" data-command="${escapeHtml(cmd.cmd || '')}">
                                        <i class="fas fa-copy"></i>
                                    </button>
                                </div>
                                <div class="command-info">
                                    <i class="fas fa-info-circle"></i> ${escapeHtml(cmd.info || 'No description available')}
                                </div>
                            </div>
                        `).join('') : '<p>No commands available.</p>'}
                    </div>
                    
                    ${methodology.tips && methodology.tips.length ? `
                        <div class="tips-section">
                            <h3><i class="fas fa-lightbulb"></i> Tips & Best Practices</h3>
                            <ul class="tips-list">
                                ${methodology.tips.map(tip => `
                                    <li><i class="fas fa-check-circle"></i> ${escapeHtml(tip)}</li>
                                `).join('')}
                            </ul>
                        </div>
                    ` : ''}
                </div>
            `;

            // Add copy functionality to buttons
            const copyButtons = container.querySelectorAll('.methodology-copy-btn');
            copyButtons.forEach(button => {
                button.addEventListener('click', () => {
                    const command = button.dataset.command;
                    copyCommand(null, command);
                });
            });
        }
